CREATE TABLE PERITO( 
    ID_PERITO  INTEGER NOT NULL ,
    NOME_PERITO VARCHAR2 (50),
    EMAIL_PERITO VARCHAR2 (50),
    ENDERECO VARCHAR2 (50),
    TELEFONE VARCHAR2 (10),
    PRIMARY KEY ( ID_PERITO )
);

CREATE TABLE PROVA( PRODUTO
	ID_PROVA INTEGER NOT NULL , 
	NOME_PROVA VARCHAR2 (70),
        NUMERO_PROCESSO VARCHAR2 (10),
        LAUDO VARCHAR2 (200),
        ID_PERITO INTEGER NOT NULL,
	PRIMARY KEY ( ID_PROVA ),
	FOREIGN KEY ( ID_PERITO ) REFERENCES PERITO( ID_PERITO )
);
